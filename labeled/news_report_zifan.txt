1. How many journalists are there?

SELECT count(*)
FROM journalist

2. List the names of journalists in ascending order of years working.

SELECT Name
FROM journalist
order by Years_working ASC

3. What are the nationalities and ages of journalists?

SELECT Nationality, Age
FROM journalist

4. Show the names of journalists with nationality "England" or "Wales".

SELECT Name
FROM journalist
Where Nationality = "England" or Nationality = "Wales"

5. What is the average years of working of journalist?

SELECT avg(Years_working)
FROM journalist

6. What is the nationality of the journalist with the largest number of years working?

SELECT Nationality
FROM journalist
order by Years_working DESC limit 1

7. Show different nationalities and the number of journalists of each nationality.

SELECT Nationality, COUNT(*)
FROM journalist
group by Nationality
 
8. Show the most common nationality of journalists.

SELECT Nationality
FROM journalist
group by Nationality
order by COUNT(*) DESC limit 1

9. Show the nations that have both journalists with more than 10 years of working and journalists with less than 3 years of working.

SELECT Nationality
from journalist
Where Years_working > 10
Intersect 
SELECT Nationality
from journalist

10. Show the dates and names of events in descending order of the attendance.

SELECT Date, Name
from event
order by Event_Attendance DESC 

11. Show names of journalists and the dates of the events they reported.

SELECT T3.Name, T2.Date
FROM news_report as T1 JOIN event as T2
on T1.Event_ID = T2.Event_ID join journalist as T3
on T1.journalist_ID = T3.journalist_ID

12. Show names of journalists and the names of the events they reported in ascending order 
of the attendance of the event.

SELECT T3.Name, T2.Name
FROM news_report as T1 JOIN event as T2
on T1.Event_ID = T2.Event_ID join journalist as T3
on T1.journalist_ID = T3.journalist_ID
order by T2.Event_Attendance ASC

13. Show the names of journalists and the number of events they reported.

SELECT T3.Name, COUNT(*)
FROM news_report as T1 JOIN event as T2
on T1.Event_ID = T2.Event_ID join journalist as T3
on T1.journalist_ID = T3.journalist_ID
group by T3.Name

14. Show the names of journalists that have reported more than one event.

SELECT T3.Name
FROM news_report as T1 JOIN event as T2
on T1.Event_ID = T2.Event_ID join journalist as T3
on T1.journalist_ID = T3.journalist_ID
group by T3.Name
Having COUNT(*) > 1

15. List the names of journalists who have not reported any event.

SELECT Name
from journalist
where journalist_ID not in (SELECT journalist_ID FROM news_report)
