1. How many buildings are there?

SELECT count(*)
FROM building

2. List the names of buildings in ascending order of number of stories.

SELECT Name
FROM building
order by Number_of_Stories ASC

3. List the addresses of buildings in descending order of completed year.

SELECT Address
FROM building
order by Completed_Year DESC

4. What are the maximum number of stories of buildings not completed in 1980?

SELECT max(Number_of_Stories)
FROM building
Where Completed_Year != "1980"

5. What is the average population for all regions?

SELECT avg(Population)
FROM region

6. What are the names of regions in ascending alphabetical order?

SELECT Name
FROM region
order by Name ASC

7. What are the capitals of regions with area bigger than 10000?

SELECT Capital
FROM region
Where Area > 10000

8. List the capital of the region with the largest population.

SELECT Capital
FROM region

9. List the names of the regions with the top 5 largest area.

SELECT Name

10. Show names of buildings and the names of regions they are in.

SELECT T1.Name, T2.Name
FROM building as T1 join region as T2 
ON T1.Region_ID = T2.Region_ID

11. Show names of regions that have more than one building.

SELECT T2.Name
FROM building as T1 join region as T2 
ON T1.Region_ID = T2.Region_ID
group by T1.Region_ID
Having COUNT(*) > 1
