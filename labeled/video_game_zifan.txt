9. List names and colleges of all players in descending order of rank of the year.

10. Please show the names of players that have played the game titled "Super Mario World".

SELECT T3.Player_name
FROM game as T1 join game_player as T2 
on T1.Game_ID = T2.Game_ID
join player as T3
on T2.Player_ID = T3.Player_ID
Where T1.Title = "Super Mario World"

11. Show the distinct developer of games played by players that go to college "Auburn".

SELECT distinct T1.Developers
FROM game as T1 join game_player as T2 
on T1.Game_ID = T2.Game_ID
join player as T3
on T2.Player_ID = T3.Player_ID
Where T3.College = "Auburn"

13. Please list the title and platform name of games.

FROM game as T1 join platform as T2

14. Please list the title of games with platforms that have market district in Asia
or USA.

SELECT T1.Title
FROM game as T1 join platform as T2 
on T1.Platform_ID = T2.Platform_ID
Where T2.Market_district = "Asia" or T2.Market_district = "USA"
