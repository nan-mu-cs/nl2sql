1. How many teachers are there?

FROM teacher

2. List the names of teachers in ascending order of age.

SELECT Name
FROM teacher
order by Age ASC

3. What are the age and hometown of teachers?

SELECT Age, Hometown
FROM teacher

4. List the name of teachers whose hometown is not "Little Lever Urban District".

SELECT Name
FROM teacher

5. Show the name of teachers aged either 32 or 33

SELECT Name
FROM teacher

6. What is the hometown of the youngest teacher?

SELECT Hometown
FROM teacher
order by Age ASC limit 1

7. Show different hometown of teachers and the number of teachers from each hometown.

SELECT Hometown, COUNT(*)
FROM teacher
group by Hometown
 
8. Please the most common hometown of teachers.

SELECT Hometown
FROM teacher
group by Hometown
order by COUNT(*) DESC limit 1

9. Show the hometowns shared by at least two teachers.

SELECT Hometown
FROM teacher
group by Hometown
Having COUNT(*) >= 2

10. Show names of teachers and the courses they are arranged to teach.

SELECT T3.Name, T2.Course
FROM course_arrange as T1 join course as T2 
ON T1.Course_ID = T2.Course_ID join teacher as T3
on T1.Teacher_ID = T3.Teacher_ID

11. Show names of teachers and the courses they are arranged to teach in ascending alphabetical order of the teacher's name.

SELECT T3.Name, T2.Course
FROM course_arrange as T1 join course as T2 
ON T1.Course_ID = T2.Course_ID join teacher as T3
on T1.Teacher_ID = T3.Teacher_ID
order by T3.Name

12. Show the name of the teacher for the math course

SELECT T3.Name
FROM course_arrange as T1 join course as T2 
ON T1.Course_ID = T2.Course_ID join teacher as T3
on T1.Teacher_ID = T3.Teacher_ID
Where T2.Course = "Math"

13. Show names of teachers and the number of courses they teach.

SELECT T3.Name, COUNT(*)
FROM course_arrange as T1 join course as T2 
ON T1.Course_ID = T2.Course_ID join teacher as T3
on T1.Teacher_ID = T3.Teacher_ID
group by T3.Name

14. Show names of teachers that teach at least two courses.

SELECT T3.Name
FROM course_arrange as T1 join course as T2 
ON T1.Course_ID = T2.Course_ID join teacher as T3
on T1.Teacher_ID = T3.Teacher_ID
group by T3.Name
Having COUNT(*) >= 2

15. List the names of teachers that have not been arranged to teach courses.

SELECT Name
from teacher
where Teacher_id not in (SELECT Teacher_id FROM course_arrange)
