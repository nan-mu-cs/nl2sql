3. Find the login name of the course author that teaches the course with name "advanced database".

SELECT T1.login_name
FROM Course_Authors_and_Tutors as T1 JOIN Courses as T2
on T1.author_id = T2.author_id
WHERE T2.course_name = "advanced database"

Find the addresses of the course authors that teach the course with name "operating system" or "data structure".

SELECT T1.address_line_1
FROM Course_Authors_and_Tutors as T1 JOIN Courses as T2
on T1.author_id = T2.author_id

Find the personal name, family name, and author ID of the course author that teaches the most number of courses.

SELECT T1.personal_name, T1.family_name, T2.author_id
FROM Course_Authors_and_Tutors as T1 JOIN Courses as T2
on T1.author_id = T2.author_id
group by T2.author_id
order by COUNT(*) DESC
LIMIT 1

Find the addresses and author IDs of the course authors that teach at least two courses.

SELECT T1.address_line_1, T2.author_id
FROM Course_Authors_and_Tutors as T1 JOIN Courses as T2
on T1.author_id = T2.author_id
group by T2.author_id
HAVING Count(*) >= 2

Find the subject IDs, names of subjects and the corresponding courses belong to each subject 
sorted by count in ascending order.

SELECT T1.subject_id, T2.subject_name, COUNT(*)
FROM Courses as T1 JOIN Subjects as T2
ON T1.subject_id = T2.subject_id
group by T1.subject_id
order by COUNT(*) ASC
