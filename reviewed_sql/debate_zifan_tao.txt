1. How many debates are there?

SELECT count(*)
FROM debate

2. List the venue of debates in ascending order of number of audience.

SELECT Venue
FROM debate
order by Num_of_Audience ASC

3. What are the date and venue of debates?

SELECT Date, Venue
FROM debate

4. List the date of debates with number of audience bigger than 150

SELECT Date
FROM debate
Where Num_of_Audience > 150

5. Show the name of people aged either 35 or 36.

SELECT Name
FROM  people
Where Age = 35 or Age = 36

6. What is the party of the youngest people?

SELECT Party
FROM people
order by Age ASC limit 1

7. Show different parties of people and the number of people in each party.

SELECT Party, COUNT(*)
FROM people
group by Party
 
8. Show the party that has the most number of people.

SELECT Party
FROM people
group by Party
order by COUNT(*) DESC limit 1

9. Show the distinct venue of debates

SELECT distinct Venue
from debate

10. Show names of people, and dates and venues of debates they are on affirmative side.

SELECT T3.Name, T2.Date, T2.Venue
FROM debate_people as T1 join debate as T2 
ON T1.Debate_ID = T2.Debate_ID join people as T3
on T1.Affirmative = T3.People_ID

11. Show names of people, and dates and venues of debates they are on negative side in ascending alphabetical order of name.

SELECT T3.Name, T2.Date, T2.Venue
FROM debate_people as T1 join debate as T2 
ON T1.Debate_ID = T2.Debate_ID join people as T3
on T1.Negative = T3.People_ID
order by T3.Name ASC

12. Show the names of people that are on affirmative side of debates with number of audience bigger than 200

SELECT T3.Name
FROM debate_people as T1 join debate as T2 
ON T1.Debate_ID = T2.Debate_ID join people as T3
on T1.Affirmative = T3.People_ID
Where T2.Num_of_Audience > 200

13. Show names of people and the number of times they have been on affirmative side of the debate

SELECT T2.Name, COUNT(*)
FROM debate_people as T1 join people as T2
on T1.Affirmative = T2.People_ID
group by T2.Name

14. Show names of people that have been on negative side of the debate at least twice.

SELECT T2.Name
FROM debate_people as T1 join people as T2
on T1.Negative = T2.People_ID
group by T2.Name
Having COUNT(*) >= 2

15. List the names of people that have not been on affirmative side of the debate.

SELECT Name
from people
where People_id not in (SELECT Affirmative FROM debate_people)
