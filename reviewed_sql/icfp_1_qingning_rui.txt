2. What are the titles of papers published by "Jeremy Gibbons"?

select t3.title
from authors as t1 join authorship as t2
on t1.authid = t2.authid
join papers as t3
on t2.paperid = t3.paperid
where  t1.fname = "Jeremy" and t1.lname = "Gibbons"

8. Find the title of papers that only have one author.

select title
from (select T1.title, count(distinct T2.authid) as author_count
from papers as T1 join authorship as T2
on t1.paperid = t2.paperid
group by t1.title)
where author_count = 1

11. What is the last name of the author that has published the most papers?

select t1.lname
from authors as t1 join authorship as t2
on t1.authid = t2.authid
join papers as t3
on t2.paperid = t3.paperid
group by t1.fname, t1.lname
order by count(*) desc
limit 1

from authorship as t1 join papers as t2
on t1.paperid = t2.paperid

from authorship)
