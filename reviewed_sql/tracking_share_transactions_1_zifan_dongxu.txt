3. Show the average amount of different transaction types.

select transaction_type_code, avg(amount_of_transaction)
from TRANSACTIONS
group by transaction_type_code

Show the maximum and minimum share count of different transaction types.

select transaction_type_code, max(share_count), min(share_count)
from TRANSACTIONS
group by transaction_type_code

Show the average share count of transactions for different investors.

select investor_id, avg(share_count)
from TRANSACTIONS
group by investor_id

Show the average share count of transactions for different investors order by average share count.

select investor_id, avg(share_count)
from TRANSACTIONS
group by investor_id
ORDER BY avg(share_count)

Show the average amount of transactions for different investors.

select investor_id, avg(amount_of_transaction)
from TRANSACTIONS
group by investor_id

Show the average amount of transactions for different lots.

select T2.lot_id, avg(amount_of_transaction)
from TRANSACTIONS as T1 JOIN Transactions_Lots as T2
on T1.transaction_id = T2.transaction_id
group by T2.lot_id

Show the average amount of transactions for different lots order by average amount of transactions.

select T2.lot_id, avg(amount_of_transaction)
from TRANSACTIONS as T1 JOIN Transactions_Lots as T2
on T1.transaction_id = T2.transaction_id
group by T2.lot_id
ORDER BY avg(amount_of_transaction)

select transaction_type_code

Show the transaction type code that occurs the most number of times.

select transaction_type_code
from TRANSACTIONS

Show the description of type of transaction that occurs the most number of times.

select T1.transaction_type_description
from Ref_Transaction_Types as T1 JOIN TRANSACTIONS as T2
on T1.transaction_type_code = T2.transaction_type_code
group by T1.transaction_type_code

Show the id and details of the investor that have the biggest number of transactions.

select T2.investor_id, T1.Investor_details
from INVESTORS as T1 JOIN TRANSACTIONS as T2
on T1.investor_id = T2.investor_id
group by T2.investor_id

Show the id and details of the investor that have the top 3 number of transactions.

select T2.investor_id, T1.Investor_details
from INVESTORS as T1 JOIN TRANSACTIONS as T2
on T1.investor_id = T2.investor_id

Show the id of the investor that have at least two transactions.

select T2.investor_id
from INVESTORS as T1 JOIN TRANSACTIONS as T2
on T1.investor_id = T2.investor_id
group by T2.investor_id
HAVING COUNT(*) >= 2

Show the ids and details of the investor that have at least two transactions with type code "SALE".

select T2.investor_id, T1.Investor_details
from INVESTORS as T1 JOIN TRANSACTIONS as T2
on T1.investor_id = T2.investor_id
WHERE T2.transaction_type_code = "SALE"
group by T2.investor_id
HAVING COUNT(*) >= 2

4. What are the dates of transactions with type code "SALE" and at least 100 share count or with type code "PUR" and at least 200 share count?

select date_of_transaction
From TRANSACTIONS
WHERE (transaction_type_code = "SALE" AND share_count >= 100) or (transaction_type_code = "PUR" AND share_count >= 200)


What are the dates of transactions with at least 100 share count and amount bigger than 100 or with at least 200 share count and amount bigger than 50?

select date_of_transaction
From TRANSACTIONS
WHERE (share_count >= 100 AND amount_of_transaction >= 100) or (share_count >= 200 AND amount_of_transaction >= 50)
